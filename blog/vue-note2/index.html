var alertComponent = function(content,callback,button) {
    if(button == undefined) {
        button = [{key:'alert',text:'确定'}]
    }
    let ErrorAlert
    if(button.length===1){
        ErrorAlert = Vue.extend({
            template: "<MyAlert1 v-if='showFlag' title='Title'" +
                " :content='alertContent' :operation='alertButton[0].text' " +
                ":clickMaskHidden='false' @cancel='alertClick' />",
            components: {MyAlert1},
            data(){return{
                alertContent:content,
                showFlag:true,
                alertButton: button
            }},
            methods: {
                alertClick(){
                    let that = this;
                    setTimeout(()=>{that.showFlag = false},100);
                    callback&&callback()
                }
            }
        })
    } else {
        ErrorAlert = Vue.extend({
            template: "<MyAlert2 v-if='showFlag' title='Title' " +
                ":cancelOperation='alertButton[0].text' @cancel='cancel'" +
                ":confirmOperate='alertButton[1].text' @confirm='confirm' />",
            components:{MyAlert2},
            data(){
                return{
                    alertContent:content,
                    alertButton:button,
                    showFlag: true
                }
            },
            methods:{
                cancel(){
                    let that = this;
                    setTimeout(function(){that.showFlag = false},100);
                    callback&&callback(this.alertButton[0].key)
                },
                confirm(){
                    let that = this;
                    setTimeout(function(){that.showFlag = false},100);
                    callback&&callback(this.alertButton[1].key)
                }
            }
        })
    }
    let ShowErrorAlert = new ErrorAlert().$mount();
    document.getElementById('app').appendChild(ShowErrorAlert.$el)
}
