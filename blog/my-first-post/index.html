先写一个blog/md的格式规范

+++ //开头
title = "标题名称"
draft = "false" //草稿
author = "么了"
categories = ["note"] //设置分类，支持数组格式
date = "2019-04-02"
description = ""
featured = "pic01.jpg" //图片地址
featuredalt = "Pic 1"
featuredpath = "date"
linktitle = ""
type = "post"
MetaDataFormat = "toml"
languageCode = "en-us"
date = "2014-09-29T21:54:53-05:00"
Tags = []

+++ //结尾
 
//+++中间的是设置内容
