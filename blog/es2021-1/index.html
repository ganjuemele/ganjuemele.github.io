逻辑分配运算符

November 12, 2021 么了

1 分钟阅读

  • Twitter

  • Facebook

  • LinkedIn

  • Email

1. ||= 或等于运算符

let a = false;
let b = true;
a ||= b //a=true

let c = 0;
let d= true;
c ||= d //c=true

let e= 1;
let f = false;
e ||= f  //e=1

a ||= b 相当于 a || (a=b)

2. &&= 且等于运算符

function() {
    let a;
    let b = 0;
    let c = 1;
    a &&= 2;
    b &&= 2;
    c &&= 2;
    return typeof a === 'undefined' && b === 0 && c === 2;
}//true

function(){
    let a;
    let i = 1;
    a &&= ++i;
    return typeof a === 'undefined' && i === 1;
}

a &&= b 相当于 a && (a=b)

3. ??= 无效赋值运算符

function(){
    let a;
    let b = 0;
    let c = 1;
    a ??= 2;
    b ??= 2;
    c ??= 2;
    return a === 2 && b === 0 && c === 1;
} //true

判断a是否为undefined || null

⚠️注意： (a=(a==2)) ||= b 左手边只能是计算出的结果，不能有算式，否则报错
