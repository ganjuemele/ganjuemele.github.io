Node.js文件系统模块

方法1：

// 引入 fs 模块
let fs = require("fs")
let fs = window.require('fs') // vue中需要在全局引入

// fs.readFile(path, cb)
fs.readFile('./src/data.json', function (err, data) {
    // data为文件内容
    if(err) {
        return console.error(err)
    }
})

// fs.writeFile('文件路径'，'写入的内容'，fn)
fs.writeFile('./src/data.json', 'TEXT', function (err, data) {
    if (err) {
        return console.error(err)
    }
})

方法2：

//需要的使用的地方 引入js文件
import xx from '../data/file.js'
// file.js
fs.readFile('./src/data.json', function (err, data) {
    if(err) {
        return console.error(err)
    }
    let dt = data.toString() // 缓存之前文件的内容
    fs.writeFile('./src/data.json', dt+'TEXT', function (err, data) {
        if (err) {
            return console.error(err)
        }
    })
})
