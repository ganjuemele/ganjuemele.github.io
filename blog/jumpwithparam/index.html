入参url 跳转链接

入参JSON Object 自动转换 key:value value只能为字符串

var jumpWithParam = function(url, obj){
    let jsonStr = JSON.stringify(obj);
    jsonStr = jsonStr.substr(1,jsonStr.length-2).replace(/"/g,'');
    let jsonList = jsonStr.split(',');
    for(let i=0;i<jsonList.length;i++){
        if(i!==0){
            url=url+'&';
        } else{
            url=url+'?';
        }
        let KVpair = jsonList[i].split(':');
        url = url + KVpair[0]+'='+KVpair[1];
    }
    console.log(url);
    my.navigateTo({
      url: url
    });
};
